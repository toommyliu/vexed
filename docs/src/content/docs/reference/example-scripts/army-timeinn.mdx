import { Code } from "@astrojs/starlight/components";

export const sampleCode = String.raw`

// if you want to complete the daily:
// cmd.accept_quest(8152..8154)
// cmd.complete_quest(8152..8154)

cmd.enable_lagkiller()
cmd.enable_infiniterange()
cmd.enable_anticounter()
cmd.goto_house()

cmd.army_set_config('army_config')
cmd.army_init()

cmd.accept_quest(8152)
cmd.accept_quest(8153)
cmd.accept_quest(8154)
cmd.equip_item("Elders' Blood")
cmd.equip_item('Scroll of Enrage')

let opts = {
    skillAction() {
      let a = [] // skill combo
      let i = 0 // current skill index
      let h // optional heal skill
      switch (this.bot.player.className) {
        case "LEGION REVENANT":
          a=[3,1,2,4,5]
          break
        case "STONECRUSHER":
          a=[1,2,3,4,5]
          break
        case "LORD OF ORDER":
          a=[1,3,4,5]
          h=2
          break
        case "ARCHPALADIN":
          a=[1,3,4,5]
          h=2
      }

      return async function() {
        for (const plyr /* string */ of this.bot.army.players) {
          const p = this.bot.world.players.get(plyr)
          if (p.isHpPercentageLessThan(70) && h) {
            await this.bot.combat.useSkill(h)
            return;
          }
        }

        await this.bot.combat.useSkill(a[i])
        i = (i + 1) % a.length;
      }
    }
  }
}

cmd.army_join('ultraezrajal','r2','Left')
cmd.set_spawn()
cmd.army_kill_for(
  'Ultra Ezrajal',
  'Ultra Ezrajal Defeated',
  1,
  true,
  opts
)
cmd.move_to_cell('Enter','Spawn')

cmd.army_join('ultrawarden','r2','Left')
cmd.set_spawn()
cmd.army_kill_for(
  'Ultra Warden',
  'Ultra Warden Defeated',
  1,
  true,
  opts
)
cmd.move_to_cell('Enter','Spawn')

cmd.army_join('ultraengineer','r2','Left')
cmd.set_spawn()
cmd.army_kill_for(
  'id.3',
  'Ultra Engineer Defeated',
  1,
  true,
  {
    ...opts,
    killPriority: ['id.1','id.2']
  }
)
cmd.move_to_cell('Enter','Spawn')
`
